apiVersion: batch/v1
kind: CronJob
metadata:
  name: flask-health-check
  labels:
    app: flask-app
spec:
  schedule: "*/5 * * * *"  # Run every 5 minutes
  successfulJobsHistoryLimit: 3
  failedJobsHistoryLimit: 1
  jobTemplate:
    spec:
      template:
        spec:
          containers:
          - name: health-checker
            image: curlimages/curl:latest
            command:
            - /bin/sh
            - -c
            - |
              echo "Running health check on Flask app..."
              response=$(curl -s -o /dev/null -w "%{http_code}" http://flask-app-service/)
              if [ "$response" = "200" ]; then
                echo "✓ Health check passed! Status: $response"
                echo "Response body:"
                curl -s http://flask-app-service/
              else
                echo "✗ Health check failed! Status: $response"
                exit 1
              fi
          restartPolicy: OnFailure
---
apiVersion: batch/v1
kind: CronJob
metadata:
  name: flask-metrics-report
  labels:
    app: flask-app
spec:
  schedule: "0 * * * *"  # Run every hour
  successfulJobsHistoryLimit: 2
  failedJobsHistoryLimit: 1
  jobTemplate:
    spec:
      template:
        spec:
          containers:
          - name: metrics-reporter
            image: curlimages/curl:latest
            command:
            - /bin/sh
            - -c
            - |
              echo "=== Flask App Metrics Report ==="
              echo "Timestamp: $(date)"
              echo "Testing app endpoint..."
              curl -v http://flask-app-service/
              echo ""
              echo "Report completed!"
          restartPolicy: OnFailure
---
